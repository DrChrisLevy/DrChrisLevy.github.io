import concurrent.futures
import json
from datetime import datetime
from typing import Any, Dict, List

import pandas as pd
from dotenv import load_dotenv
from openai import OpenAI

load_dotenv()

client = OpenAI()


def get_system_prompt(todays_date=datetime.now().strftime("%Y-%m-%d")):
    return f"You are a helpful assistant. Todays date is {todays_date}"


def get_sales_forecast(query, todays_date=datetime.now().strftime("%Y-%m-%d")):
    res = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[
            {"role": "system", "content": get_system_prompt(todays_date)},
            {"role": "user", "content": query},
        ],
        tools=[
            {
                "type": "function",
                "function": {
                    "name": "forecast_sales",
                    "description": "Get the sales forecast between two dates.",
                    "parameters": {
                        "type": "object",
                        "properties": {
                            "start_date": {
                                "type": "string",
                                "description": "The start date of the forecast using the format YYYY-MM-DD.",
                            },
                            "end_date": {
                                "type": "string",
                                "description": "The end date of the forecast using the format YYYY-MM-DD.",
                            },
                        },
                        "required": ["start_date", "end_date"],
                        "additionalProperties": False,
                    },
                },
            }
        ],
    )
    return json.loads(res.choices[0].message.tool_calls[0].function.arguments)


def run_experiment(test_cases: List[Dict[str, Any]]) -> pd.DataFrame:
    def process_case(case):
        query = case["query"]
        todays_date = case["todays_date"]
        expected = case["expected"]

        actual = get_sales_forecast(query, todays_date)

        return {
            "query": query,
            "todays_date": todays_date,
            "expected_start": expected["start_date"],
            "expected_end": expected["end_date"],
            "actual_start": actual["start_date"],
            "actual_end": actual["end_date"],
            "start_correct": expected["start_date"] == actual["start_date"],
            "end_correct": expected["end_date"] == actual["end_date"],
        }

    with concurrent.futures.ThreadPoolExecutor(max_workers=10) as executor:
        results = list(executor.map(process_case, test_cases))

    return pd.DataFrame(results)
