load_dotenv()


TOGETHER_AI_BASE_URL = "https://api.together.xyz/v1"
TOGETHER_API_KEY = os.environ["TOGETHER_API_KEY"]


class Clients(Enum):
    TOGETHER_AI = "TOGETHER_AI"


class OpenAIChatCompletion:
    clients: Dict = dict()

    @classmethod
    def _load_client(
        cls,
        client_name: Optional[Literal[Clients.TOGETHER_AI]] = None,
    ) -> OpenAI:
        """
        By default, will connect directly to OpenAI API. However, there is the option to connect
        to endpoints which provide compatibility for the OpenAI API standard. For example, to
        connect to together.ai.
        """
        if client_name == Clients.TOGETHER_AI:
            base_url = TOGETHER_AI_BASE_URL
            api_key = TOGETHER_API_KEY
        else:
            # will auto connect to the OpenAI API using the OPENAI_API_KEY in the env.
            base_url = None
            api_key = None
        client_key = (base_url, api_key)
        if OpenAIChatCompletion.clients.get(client_key) is None:
            OpenAIChatCompletion.clients[client_key] = OpenAI(base_url=base_url, api_key=api_key)
        return OpenAIChatCompletion.clients[client_key]

    def __call__(
        self,
        model: str,
        messages: list,
        max_retries: int = 2,
        timeout: Union[float, Timeout, None] = None,
        client_name: Optional[Literal[Clients.TOGETHER_AI]] = None,
        **kwargs: Any,
    ) -> Union[ChatCompletion, Stream[ChatCompletionChunk]]:
        # https://platform.openai.com/docs/api-reference/chat/create
        # https://github.com/openai/openai-python
        client = self._load_client(client_name)
        if not timeout:
            timeout = client.timeout
        return client.with_options(max_retries=max_retries, timeout=timeout).chat.completions.create(model=model, messages=messages, **kwargs)

To solve the tasks above, which involve analyzing arrangements and sequences based on given conditions, several reasoning modules are particularly relevant:

1. **Critical Thinking**: This involves analyzing the information presented and making logical deductions about the arrangement of items based on the conditions. It's crucial for questioning assumptions and ensuring that each deduction logically follows from the given information.

2. **How can I simplify the problem so that it is easier to solve?**: Breaking down each condition and using it to simplify the overall problem into smaller, more manageable parts can make it easier to reach a solution. 

3. **What is the core issue or problem that needs to be addressed?**: Identifying the main task in each scenario (arranging objects or determining sequences based on given criteria) is essential before attempting to solve it.

4. **How can I break down this problem into smaller, more manageable parts?**: By taking each piece of information given about the sequence and using it to make individual determinations about the arrangement, the problem becomes more approachable.

5. **What is the best way to modify this current best solution, given what you know about these kinds of problem specification?**: As you build or deduce the arrangement, continually adjust your model based on each new piece of information or rule to arrive at the correct outcome.

6. **Let’s think step by step. / Let’s make a step by step plan and implement it with good notion and explanation.**: This approach is essential for systematically applying each condition to deduce the correct order or arrangement of the items. 

These modules emphasize logical deduction, breaking problems into smaller parts, critical and step-by-step thinking, all of which are key to solving logical arrangement and sequence problems as seen in the tasks.

Rephrased and specified for each reasoning module:

1. **Logical Analysis and Deduction**: Focus on analyzing the provided conditions to logically deduce the correct arrangement or sequence. This module involves critically examining assumptions to ensure that deductions are firmly rooted in the provided information, crucial for sequencing and arrangement tasks.

2. **Decomposition of Conditions**: Tackle complex problems by breaking down each given condition to simplify the overall problem. This approach makes the task more manageable by dividing it into smaller, solvable parts, each of which contributes to constructing the final solution.

3. **Identification of the Primary Challenge**: Distill down the essence of the problem by pinpointing the main objective, whether it's deducing the correct sequence or arranging objects according to specific criteria. Understanding the core challenge is the first step in crafting an effective solution strategy.

4. **Segmentation Strategy**: Utilize each piece of given information to segment the problem into more manageable portions. By addressing each condition separately, you create a clearer pathway through which the overall problem can be systematically solved.

5. **Iterative Refinement**: With the initial arrangement or sequence in place, methodically refine your solution by incorporating new information or constraints as they are understood. This adaptive approach allows for continual improvement and adjustment of the solution until it aligns perfectly with the provided specifications.

6. **Step-by-Step Execution**: Emphasize a sequential approach to applying conditions, gradually building up to the correct arrangement or sequence. This methodical tactic ensures that each step is logically sound and well-explained, leading to a robust and understandable solution process.

By focusing these reasoning modules more directly on the challenges presented by tasks involving arrangements and sequences, solvers are better equipped to navigate the logical intricacies and arrive at accurate solutions.
